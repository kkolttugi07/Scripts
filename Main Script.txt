local LocalPlr = lib:Tab("Local Player")

LocalPlr:Toggle("Auto Refresh",function(v)
    getgenv().spw = v
    while wait() do
    if not getgenv().spw then return end
    if game.Players.LocalPlayer.Character.Humanoid.Health < 1 and swit == false then
        poi = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame
        swit = true
    end

    if swit == true then
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = poi
        end
    game.Players.LocalPlayer.CharacterAdded:connect(function()
        wait(0.6)
        swit = false
    end)
end
end)
